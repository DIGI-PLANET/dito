DITO API Documentation

Transform human potential into digital assets with our comprehensive API for talent discovery and Soul NFT management.

🌟 What is DITO?

DITO is a revolutionary platform that helps people discover their hidden talents through AI and mint them as Soul NFTs on Solana blockchain. Our API enables developers to build amazing talent-focused applications.

🔮 For Developers

Build applications that help users discover and develop their talents with our powerful API and SDK.

🤖 For AI Agents

Natural language interfaces and automation-friendly endpoints perfect for AI-powered applications.

🚀 Quick Start

Base URL: https://gateway.dito.guru (Production) | http://localhost:8099 (Development)

📦 Installation

# Install DITO SDK
npm install dito-sdk

# Or with yarn
yarn add dito-sdk

💻 Basic Usage

// Import the SDK
import { DitoClient, DitoAgent } from 'dito-sdk';

// Initialize client
const dito = new DitoClient('https://gateway.dito.guru', 'your-auth-token');

// For AI Agents - natural language interface
const agent = new DitoAgent(dito);

// Basic API calls
const soul = await dito.getSoul('user-soul-id');
const balance = await dito.getEmberBalance();
const talents = await dito.getTalents();

// AI-friendly natural language
const statusMessage = await agent.check2FAStatus();
const soulInfo = await agent.showSoulStatus('roy-soul');
console.log(statusMessage); // "2FA가 활성화되어 있습니다 (사용자: Roy)"

🔐 Authentication

All API requests require authentication. DITO uses token-based authentication with mandatory 2FA for enhanced security.

🔑 Authentication Headers

    'Authorization': 'Bearer your-auth-token-here',
    'Content-Type': 'application/json',
    'User-Agent': 'YourApp/1.0'
};

DITO API is designed to be AI Agent-friendly. Here are comprehensive examples for natural language interactions and automation.

🗣️ Natural Language Processing

import { DitoClient, DitoAgent } from 'dito-sdk';

// Initialize for AI Agent use
const dito = new DitoClient('https://gateway.dito.guru', process.env.DITO_TOKEN);
const agent = new DitoAgent(dito);

// Natural language intent processing
async function processUserIntent(userMessage) {
    const intent = analyzeIntent(userMessage);
    
    switch (intent.type) {
        case 'check_status':
            if (intent.includes('2fa') || intent.includes('보안')) {
                return await agent.check2FAStatus();
            }
            if (intent.includes('soul') || intent.includes('영혼')) {
                return await agent.showSoulStatus(intent.soulId);
            }
            if (intent.includes('ember') || intent.includes('잔액')) {
                return await agent.checkEmberBalance();
            }
            break;
            
        case 'get_recommendations':
            if (intent.includes('talent') || intent.includes('재능')) {
                return await agent.suggestTalents();
            }
            break;
            
        case 'update_progress':
            if (intent.includes('progress') || intent.includes('진행')) {
                const result = await dito.addTalentProgress(
                    intent.talentId, 
                    intent.progressAmount, 
                    intent.notes
                );
                return `✅ 진행상황 업데이트 완료! ${result.data.ember_gained} Ember 획득`;
            }
            break;
    }
    
    return "죄송합니다. 요청을 이해하지 못했습니다. 다시 말씀해 주세요.";
}

// Example usage with various user inputs
const responses = await Promise.all([
    processUserIntent("내 2FA 상태가 어떻게 되나?"),
    processUserIntent("Ember 잔액 확인해줘"), 
    processUserIntent("새로운 재능 추천받고 싶어"),
    processUserIntent("프로그래밍 실력이 10% 늘었어")
]);

console.log(responses);
// Output: Natural Korean responses for each intent

⚙️ AI Automation Workflows

// Automated talent development assistant
class TalentDevelopmentBot {
    constructor(ditoClient) {
        this.dito = ditoClient;
        this.agent = new DitoAgent(ditoClient);
    }

    // Daily routine: check progress and provide suggestions
    async dailyCheckIn(soulId) {
        try {
            // Get user's Soul status
            const soul = await this.dito.getSoul(soulId);
            const talents = await this.dito.getTalents();
            const recommendations = await this.dito.getTalentRecommendations();
            
            // Analyze progress
            const stagnantTalents = talents.data.filter(talent => {
                const daysSinceProgress = this.daysSince(talent.last_progress);
                return daysSinceProgress > 7; // No progress for a week
            });
            
            // Generate personalized report
            let report = `🌟 ${soul.data.seeker_name}님의 일일 점검 보고서\n\n`;
            
            // Current status
            report += `📊 현재 상태:\n`;
            report += `- Ember: ${soul.data.ember_points}\n`;
            report += `- 활성 재능: ${soul.data.active_talents}개\n`;
            report += `- 레벨: ${soul.data.current_level}\n\n`;
            
            // Stagnant talents alert
            if (stagnantTalents.length > 0) {
                report += `⚠️ 관심이 필요한 재능들:\n`;
                stagnantTalents.forEach(talent => {
                    report += `- ${talent.name}: ${this.daysSince(talent.last_progress)}일 동안 진행 없음\n`;
                });
                report += `\n`;
            }
            
            // New recommendations
            report += `💡 오늘의 추천 재능:\n`;
            recommendations.data.recommendations.slice(0, 2).forEach(rec => {
                report += `- ${rec.name}: ${rec.description}\n`;
            });
            
            // Next milestone
            if (soul.data.next_milestone) {
                report += `\n🎯 다음 목표: ${soul.data.next_milestone}`;
            }
            
            return report;
            
        } catch (error) {
            return `오류가 발생했습니다: ${error.message}`;
        }
    }

    // Automated progress tracking from external sources
    async trackProgressFromGitHub(soulId, githubActivity) {
        const progressUpdates = [];
        
        for (const activity of githubActivity) {
            if (activity.type === 'PushEvent') {
                // Award progress for code commits
                const result = await this.dito.addTalentProgress(
                    'talent-programming',
                    5, // 5% progress for each significant commit
                    `GitHub commit: ${activity.payload.commits[0].message}`
                );
                progressUpdates.push(result);
            }
        }
        
        return progressUpdates;
    }

    // Smart ember spending recommendations
    async optimizeEmberSpending(soulId) {
        const balance = await this.dito.getEmberBalance();
        const soul = await this.dito.getSoul(soulId);
        
        const recommendations = [];
        
        if (balance.data.ember_balance > 1000) {
            recommendations.push({
                type: 'arena_entry',
                message: '큰 Arena 이벤트 참가 추천 (고수익 잠재력)',
                cost: 500,
                expected_return: 2000
            });
        }
        
        if (soul.data.conviction_level < 50) {
            recommendations.push({
                type: 'talent_boost',
                message: '재능 부스터 구매 추천 (확신 레벨 향상)',
                cost: 200,
                benefit: 'conviction_level +20'
            });
        }
        
        return recommendations;
    }
    
    daysSince(dateString) {
        const date = new Date(dateString);
        const now = new Date();
        return Math.floor((now - date) / (1000 * 60 * 60 * 24));
    }
}

// Usage example
const bot = new TalentDevelopmentBot(dito);
const dailyReport = await bot.dailyCheckIn('roy-soul-123');
console.log(dailyReport);

// Comprehensive monitoring system for talent progression
class DitoAnalyticsAgent {
    constructor(ditoClient) {
        this.dito = ditoClient;
    }

    // Real-time talent ecosystem monitoring
    async generateEcosystemReport() {
        try {
            const analytics = await this.dito.getAnalytics();
            const souls = await this.dito.getSouls();
            const arenaEvents = await this.dito.getArenaEvents();
            
            // Calculate key metrics
            const totalActiveUsers = souls.data.length;
            const totalEmberInCirculation = souls.data.reduce((sum, soul) => sum + soul.ember_points, 0);
            const averageProgressRate = this.calculateAverageProgress(souls.data);
            const liveArenaCount = arenaEvents.data.filter(event => event.status === 'open').length;
            
            // Generate insights
            const insights = {
                ecosystem_health: this.assessEcosystemHealth(analytics.data),
                growth_trends: this.identifyGrowthTrends(souls.data),
                ember_economy: this.analyzeEmberEconomy(totalEmberInCirculation, totalActiveUsers),
                arena_participation: this.analyzeArenaEngagement(arenaEvents.data),
                recommendations: this.generateSystemRecommendations({
                    totalActiveUsers,
                    averageProgressRate,
                    liveArenaCount
                })
            };
            
            return this.formatAnalyticsReport(insights);
            
        } catch (error) {

        const soul = await this.dito.getSoul(soulId);

        };

        const timeToMastery = this.estimateTimeToMastery(factors);

        return {

// Robust error handling example
async function safeApiCall(apiFunction) {
    try {
        const result = await apiFunction();
        
        if (!result.success) {
            // Handle API-level errors
            switch (result.error.code) {
                case 'invalid_token':
                    await refreshAuthToken();
                    return await apiFunction(); // Retry once
                    
                case 'rate_limit_exceeded':
                    const retryAfter = result.error.details?.retry_after || 60;
                    await sleep(retryAfter * 1000);
                    return await apiFunction(); // Retry after delay
                    
                case 'soul_not_found':
                    throw new Error('Soul not found. Please check the Soul ID.');
                    
                case 'insufficient_ember':
                    throw new Error(`Not enough Ember. Current: ${result.error.details?.current_balance}`);
                    
                case '2fa_required':
                    const code = await prompt2FACode();
                    return await apiFunction(code);
                    
                default:
                    throw new Error(`API Error: ${result.message}`);
            }
        }
        
        return result.data;
        
    } catch (networkError) {
        // Handle network/connection errors
        if (networkError.name === 'AbortError') {
            throw new Error('Request timed out. Please try again.');
        }
        
        if (!navigator.onLine) {
            throw new Error('No internet connection. Please check your network.');
        }
        
        throw new Error(`Network error: ${networkError.message}`);
    }
}

// Usage with automatic retry logic
async function robustTalentUpdate(talentId, progress) {
    const maxRetries = 3;
    let lastError;
    
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
            return await safeApiCall(() => 
                dito.addTalentProgress(talentId, progress)
            );
        } catch (error) {
            lastError = error;
            
            if (attempt < maxRetries) {
                const delay = Math.pow(2, attempt) * 1000; // Exponential backoff
                console.log(`Attempt ${attempt} failed, retrying in ${delay}ms...`);
                await sleep(delay);
            }
        }
    }
    
    throw lastError; // All retries failed
}

function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
}
